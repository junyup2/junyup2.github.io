            return False

        if pos == 1:
            newNode.next = self.head
            self.head = newNode

        else:
            if pos == self.nodeCount + 1:
                prev = self.tail

                prev = self.getAt(pos - 1)

            self.tail = newNode

        self.nodeCount += 1
        return True
